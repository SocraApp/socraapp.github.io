   237‚Üí   237‚Üí  renderLine(line, lineIndex, lineStart, cursorPos) {
   238‚Üí   238‚Üí    // Heading: # text
   239‚Üí   239‚Üí    const headingMatch = line.match(/^(#{1,6})\s(.*)$/);
   240‚Üí   240‚Üí    if (headingMatch) {
   241‚Üí   241‚Üí      const level = headingMatch[1].length;
   242‚Üí   242‚Üí      const prefixLen = headingMatch[1].length + 1;
   243‚Üí   243‚Üí      const content = headingMatch[2];
   244‚Üí   244‚Üí      let html = this.mkSyntax('heading', lineIndex, headingMatch[1] + ' ', lineStart, lineStart + prefixLen, cursorPos);
   245‚Üí   245‚Üí      html += `<span class="md-h${level}">`;
   246‚Üí   246‚Üí      html += this.renderInline(content, lineIndex, lineStart + prefixLen, cursorPos);
   247‚Üí   247‚Üí      html += '</span>';
   248‚Üí   248‚Üí      return html;
   249‚Üí   249‚Üí    }
   250‚Üí   250‚Üí
   251‚Üí   251‚Üí    // Horizontal rule: --- or *** or ___
   252‚Üí   252‚Üí    if (line.match(/^(-{3,}|\*{3,}|_{3,})\s*$/)) {

   254‚Üí   254‚Üí      html += '<hr>';
   255‚Üí   255‚Üí      return html;
   256‚Üí   256‚Üí    }
   257‚Üí   257‚Üí
   258‚Üí   258‚Üí    // Blockquote: > text
   259‚Üí   259‚Üí    if (line.match(/^>\s/)) {
   260‚Üí   260‚Üí      let html = this.mkSyntax('blockquote', lineIndex, '> ', lineStart, lineStart + 2, cursorPos);
   261‚Üí   261‚Üí      html += '<blockquote class="md-blockquote">';
   262‚Üí   262‚Üí      html += this.renderInline(line.substring(2), lineIndex, lineStart + 2, cursorPos);
   263‚Üí   263‚Üí      html += '</blockquote>';
   264‚Üí   264‚Üí      return html;
   265‚Üí   265‚Üí    }
   266‚Üí   266‚Üí
   267‚Üí   267‚Üí    // Unordered list: - item or * item or + item
   268‚Üí   268‚Üí    const ulMatch = line.match(/^([-*+])\s(.*)$/);
   269‚Üí   269‚Üí    if (ulMatch) {
   270‚Üí   270‚Üí      let html = this.mkSyntax('list', lineIndex, ulMatch[1] + ' ', lineStart, lineStart + 2, cursorPos);
   271‚Üí   271‚Üí      html += '<span class="md-list-item">';
   272‚Üí   272‚Üí      html += this.renderInline(ulMatch[2], lineIndex, lineStart + 2, cursorPos);
   273‚Üí   273‚Üí      html += '</span>';
   274‚Üí   274‚Üí      return html;
   275‚Üí   275‚Üí    }
   276‚Üí   276‚Üí
   277‚Üí   277‚Üí    // Ordered list: 1. item
   278‚Üí   278‚Üí    const olMatch = line.match(/^(\d+\.)\s(.*)$/);
   279‚Üí   279‚Üí    if (olMatch) {
   280‚Üí   280‚Üí      const prefixLen = olMatch[1].length + 1;

   542‚Üí   542‚Üí

   549‚Üí   549‚Üí
   550‚Üí   550‚Üí    while (node && node !== this.editorEl) {
   551‚Üí   551‚Üí      if (node.nodeType === Node.ELEMENT_NODE) {
   552‚Üí   552‚Üí        if (node.dataset && node.dataset.line !== undefined) {

   554‚Üí   554‚Üí        }

   557‚Üí   557‚Üí        }

   559‚Üí   559‚Üí      node = node.parentNode;
   560‚Üí   560‚Üí    }
   561‚Üí   561‚Üí
   562‚Üí   562‚Üí    if (lineIndex === -1) {
   563‚Üí   563‚Üí      // Try finding .md-line parent
   564‚Üí   564‚Üí      const lineEl = range.startContainer.parentElement?.closest?.('.md-line');

   593‚Üí   593‚Üí  updateCursorPosFromDOM() {
   594‚Üí   594‚Üí    // Calculate raw markdown position from DOM cursor position
   595‚Üí   595‚Üí    const sel = window.getSelection();
   596‚Üí   596‚Üí    if (!sel.rangeCount) return;
   597‚Üí   597‚Üí
   598‚Üí   598‚Üí    const range = sel.getRangeAt(0);
   599‚Üí   599‚Üí    const preRange = range.cloneRange();
   600‚Üí   600‚Üí    preRange.selectNodeContents(this.editorEl);
   601‚Üí   601‚Üí    preRange.setEnd(range.startContainer, range.startOffset);
   602‚Üí   602‚Üí
   603‚Üí   603‚Üí    // Count text content before cursor (including text in md-syntax spans)
   604‚Üí   604‚Üí    const textBefore = preRange.toString();
   605‚Üí   605‚Üí    // But we need to account for line breaks from <div class="md-line"> elements
   606‚Üí   606‚Üí    // preRange.toString() doesn't add \n for div boundaries
   607‚Üí   607‚Üí    // We need a custom count
   608‚Üí   608‚Üí
   609‚Üí   609‚Üí    let offset = 0;
   610‚Üí   610‚Üí    let found = false;
   611‚Üí   611‚Üí
   612‚Üí   612‚Üí    const countText = (container) => {
   613‚Üí   613‚Üí      if (found) return;
   614‚Üí   614‚Üí      for (const child of container.childNodes) {
   615‚Üí   615‚Üí        if (found) return;
   616‚Üí   616‚Üí
   617‚Üí   617‚Üí        if (child === range.startContainer || child.contains(range.startContainer)) {
   618‚Üí   618‚Üí          if (child.nodeType === Node.TEXT_NODE) {
   619‚Üí   619‚Üí            offset += (child === range.startContainer) ? range.startOffset : child.textContent.length;
   620‚Üí   620‚Üí            found = true;
   621‚Üí   621‚Üí            return;
   622‚Üí   622‚Üí          } else if (child.nodeType === Node.ELEMENT_NODE) {
   623‚Üí   623‚Üí            if (child.contains(range.startContainer)) {
   624‚Üí   624‚Üí              countText(child);
   625‚Üí   625‚Üí              return;
   626‚Üí   626‚Üí            }
   627‚Üí   627‚Üí          }
   628‚Üí   628‚Üí        }
   629‚Üí   629‚Üí
   630‚Üí   630‚Üí        if (child.nodeType === Node.TEXT_NODE) {
   631‚Üí   631‚Üí          offset += child.textContent.length;
   632‚Üí   632‚Üí        } else if (child.nodeType === Node.ELEMENT_NODE) {
   633‚Üí   633‚Üí          if (child.classList?.contains('md-line')) {
   634‚Üí   634‚Üí            // Add newline between lines
   635‚Üí   635‚Üí            if (child !== container.firstChild) offset += 1;
   636‚Üí   636‚Üí          }
   637‚Üí   637‚Üí          countText(child);
   638‚Üí   638‚Üí        }
   639‚Üí   639‚Üí      }
   640‚Üí   640‚Üí    };
   641‚Üí   641‚Üí
   642‚Üí   642‚Üí    countText(this.editorEl);
   643‚Üí   643‚Üí    if (found) {
   644‚Üí   644‚Üí      this.cursorPos = offset;
   645‚Üí   645‚Üí    }
   646‚Üí   646‚Üí  }
